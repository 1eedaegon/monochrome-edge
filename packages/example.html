ðŸ’» Usage Examples

import { ThemeManager, Button, Toast } from '@monochrome-edge/core';

// Initialize theme
const theme = new ThemeManager({
  defaultTheme: 'warm',
  defaultMode: 'light'
});

// Create components
const button = Button.create({
  variant: 'primary',
  text: 'Show Toast',
  onClick: () => {
    Toast.success('Hello from Monochrome!');
  }
});

// Mount to DOM
document.body.appendChild(button.getElement());
 
import { ThemeProvider, Button, useToast } from '@monochrome-edge/react';

function App() {
  const toast = useToast();

  return (
    <ThemeProvider defaultTheme="warm">
      <Button
        variant="primary"
        onClick={() => toast.success('Hello from React!')}
      >
        Show Toast
      </Button>
    </ThemeProvider>
  );
}
 
<template>
  <MonochromeProvider :theme="warm">
    <MButton
      variant="primary"
      @click="showToast"
    >
      Show Toast
    </MButton>
  </MonochromeProvider>
</template>

<script setup>
import { MButton, useToast } from '@monochrome-edge/vue';

const toast = useToast();
const showToast = () => toast.success('Hello from Vue!');
</script>
 
<!-- Import the web components -->
<script type="module">
  import '@monochrome-edge/web';
</script>

<!-- Use the components -->
<monochrome-button
  variant="primary"
  onclick="showToast()"
>
  Show Toast
</monochrome-button>

<script>
function showToast() {
  document.dispatchEvent(
    new CustomEvent('monochrome:toast', {
      detail: { type: 'success', message: 'Hello!' }
    })
  );
}
</script>

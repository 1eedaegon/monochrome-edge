import React, { useState, ChangeEvent, FormEvent } from 'react';

interface FormData {
  email: string;
  password: string;
  remember: boolean;
}

export const LoginForm: FC = () => {
  const [formData, setFormData] = useState<FormData>({
    email: '',
    password: '',
    remember: false
  });

  const handleChange = (e: ChangeEvent<HTMLInputElement>): void => {
    const { name, value, type, checked } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: type === 'checkbox' ? checked : value
    }));
  };

  const handleSubmit = (e: FormEvent<HTMLFormElement>): void => {
    e.preventDefault();
    console.log('Form submitted:', formData);
  };

  return (
    <form onSubmit={handleSubmit} className="form">
      <div className="form-group">
        <label htmlFor="email" className="label">Email</label>
        <input
          type="email"
          id="email"
          name="email"
          className="input"
          value={formData.email}
          onChange={handleChange}
          required
        />
      </div>
      <div className="form-group">
        <label htmlFor="password" className="label">Password</label>
        <input
          type="password"
          id="password"
          name="password"
          className="input"
          value={formData.password}
          onChange={handleChange}
          required
        />
      </div>
      <div className="form-group">
        <label className="checkbox">
          <input
            type="checkbox"
            name="remember"
            checked={formData.remember}
            onChange={handleChange}
          />
          <span className="checkbox-mark"></span>
          Remember me
        </label>
      </div>
      <Button type="submit" variant="primary">
        Sign In
      </Button>
    </form>
  );
};

type ButtonVariant = 'primary' | 'secondary' | 'ghost' | 'outline' | 'danger';
type ButtonSize = 'small' | 'medium' | 'large';

const props = defineProps({
  variant: {
    type: String as PropType<ButtonVariant>,
    default: 'primary'
  },
  size: {
    type: String as PropType<ButtonSize>,
    default: 'medium'
  },
  loading: {
    type: Boolean,
    default: false
  },
  disabled: {
    type: Boolean,
    default: false
  }
});

const emit = defineEmits<{
  click: [event: MouseEvent]
}>();

const buttonClasses = computed(() => {
  const sizeClass = props.size === 'small' ? 'btn-small' :
                    props.size === 'large' ? 'btn-large' : '';
  return [
    'btn',
    `btn-${props.variant}`,
    sizeClass,
    { 'loading': props.loading }
  ].filter(Boolean).join(' ');
});

const handleClick = (event: MouseEvent): void => {
  if (!props.disabled && !props.loading) {
    emit('click', event);
  }
};

Composable for Theme
// useTheme.ts
import { ref, Ref, watch } from 'vue';

type Theme = 'warm' | 'cold';
type Mode = 'light' | 'dark';

interface ThemeConfig {
  theme: Ref<Theme>;
  mode: Ref<Mode>;
  toggleTheme: () => void;
  toggleMode: () => void;
  setTheme: (theme: Theme) => void;
  setMode: (mode: Mode) => void;
}

export function useTheme(): ThemeConfig {
  const theme = ref<Theme>('warm');
  const mode = ref<Mode>('light');

  const applyTheme = (): void => {
    const themeLink = document.getElementById('theme-link') as HTMLLinkElement;
    if (themeLink) {
      themeLink.href = `ui/tokens/${theme.value}-theme.css`;
    }
    document.documentElement.setAttribute('data-theme', mode.value);
  };

  watch([theme, mode], applyTheme, { immediate: true });

  const toggleTheme = (): void => {
    theme.value = theme.value === 'warm' ? 'cold' : 'warm';
  };

  const toggleMode = (): void => {
    mode.value = mode.value === 'light' ? 'dark' : 'light';
  };

  const setTheme = (newTheme: Theme): void => {
    theme.value = newTheme;
  };

  const setMode = (newMode: Mode): void => {
    mode.value = newMode;
  };

  return {
    theme,
    mode,
    toggleTheme,
    toggleMode,
    setTheme,
    setMode
  };
}

// MonochromeButton.ts
export class MonochromeButton extends HTMLElement {
  private _variant: string = 'primary';
  private _size: string = 'medium';
  private _loading: boolean = false;

  static get observedAttributes(): string[] {
    return ['variant', 'size', 'loading', 'disabled'];
  }

  constructor() {
    super();
    this.attachShadow({ mode: 'open' });
  }

  connectedCallback(): void {
    this.render();
  }

  attributeChangedCallback(name: string, oldValue: string | null, newValue: string | null): void {
    if (oldValue !== newValue) {
      switch (name) {
        case 'variant':
          this._variant = newValue || 'primary';
          break;
        case 'size':
          this._size = newValue || 'medium';
          break;
        case 'loading':
          this._loading = newValue === 'true';
          break;
      }
      this.render();
    }
  }

  private render(): void {
    if (!this.shadowRoot) return;

    const sizeClass = this._size === 'small' ? 'btn-small' :
                      this._size === 'large' ? 'btn-large' : '';
    const classes = `btn btn-${this._variant} ${sizeClass} ${this._loading ? 'loading' : ''}`.trim();

    this.shadowRoot.innerHTML = `
      <link rel="stylesheet" href="/ui/monochrome-edge.css">
      <button class="${classes}" ${this.hasAttribute('disabled') ? 'disabled' : ''}>
        <slot></slot>
      </button>
    `;
  }

  // Public methods
  public setLoading(loading: boolean): void {
    this._loading = loading;
    this.setAttribute('loading', loading.toString());
  }

  public click(): void {
    const button = this.shadowRoot?.querySelector('button');
    button?.click();
  }
}

// Register the custom element
customElements.define('monochrome-button', MonochromeButton);

// TypeScript declaration for JSX
declare global {

      'monochrome-button': {

        size?: 'small' | 'medium' | 'large';
        loading?: boolean;
        disabled?: boolean;
      };
    }
  }
}

// types.ts
export interface ComponentOptions {
  className?: string;
  id?: string;
  attributes?: Record<string, string>;
}

export interface ButtonOptions extends ComponentOptions {
  variant?: 'primary' | 'secondary' | 'ghost' | 'outline' | 'danger';
  size?: 'small' | 'medium' | 'large';
  loading?: boolean;
  disabled?: boolean;
  onClick?: (event: MouseEvent) => void;
}

// components.ts
export class MonochromeComponents {
  static createElement<K extends keyof HTMLElementTagNameMap>(
    tag: K,
    options: ComponentOptions = {},
    children?: (Node | string)[]
  ): HTMLElementTagNameMap[K] {
    const element = document.createElement(tag);

    if (options.className) {
      element.className = options.className;
    }

    if (options.id) {
      element.id = options.id;
    }

    if (options.attributes) {
      Object.entries(options.attributes).forEach(([key, value]) => {
        element.setAttribute(key, value);
      });
    }

    if (children) {
      children.forEach(child => {
        if (typeof child === 'string') {
          element.appendChild(document.createTextNode(child));
        } else {
          element.appendChild(child);
        }
      });
    }

    return element;
  }

  static button(text: string, options: ButtonOptions = {}): HTMLButtonElement {
    const {
      variant = 'primary',
      size = 'medium',
      loading = false,
      disabled = false,
      onClick,
      ...componentOptions
    } = options;

    const sizeClass = size === 'small' ? 'btn-small' :
                      size === 'large' ? 'btn-large' : '';
    const classes = [
      'btn',
      `btn-${variant}`,
      sizeClass,
      loading ? 'loading' : '',
      componentOptions.className || ''
    ].filter(Boolean).join(' ');

    const button = this.createElement('button', {
      ...componentOptions,
      className: classes,
      attributes: {
        ...componentOptions.attributes,
        ...(disabled ? { disabled: 'true' } : {})
      }
    }, [text]);

    if (onClick) {
      button.addEventListener('click', onClick);
    }

    return button;
  }

  static input(options: {
    type?: string;
    placeholder?: string;
    value?: string;
    onChange?: (value: string) => void;
  } & ComponentOptions = {}): HTMLInputElement {
    const {
      type = 'text',
      placeholder = '',
      value = '',
      onChange,
      ...componentOptions
    } = options;

    const input = this.createElement('input', {
      ...componentOptions,
      className: `input ${componentOptions.className || ''}`.trim(),
      attributes: {
        ...componentOptions.attributes,
        type,
        placeholder,
        value
      }
    });

    if (onChange) {
      input.addEventListener('input', (e) => {
        onChange((e.target as HTMLInputElement).value);
      });
    }

    return input;
  }

  static card(
    content: (Node | string)[],
    options: ComponentOptions = {}
  ): HTMLDivElement {
    return this.createElement('div', {
      ...options,
      className: `card ${options.className || ''}`.trim()
    }, content);
  }
}

Form Builder
interface FormField {
  name: string;
  label: string;
  type: 'text' | 'email' | 'password' | 'textarea' | 'select' | 'checkbox';
  required?: boolean;
  options?: { value: string; label: string }[];
  validation?: (value: any) => string | null;
}

class FormBuilder {
  private fields: FormField[] = [];
  private values: Record<string, any> = {};
  private errors: Record<string, string> = {};

  addField(field: FormField): this {
    this.fields.push(field);
    this.values[field.name] = '';
    return this;
  }

  setValue(name: string, value: any): void {
    this.values[name] = value;
    this.validateField(name);
  }

  private validateField(name: string): boolean {
    const field = this.fields.find(f => f.name === name);
    if (!field) return true;

    if (field.required && !this.values[name]) {
      this.errors[name] = `${field.label} is required`;
      return false;
    }

    if (field.validation) {
      const error = field.validation(this.values[name]);
      if (error) {
        this.errors[name] = error;
        return false;
      }
    }

    delete this.errors[name];
    return true;
  }

  validate(): boolean {
    let isValid = true;
    this.fields.forEach(field => {
      if (!this.validateField(field.name)) {
        isValid = false;
      }
    });
    return isValid;
  }

  render(container: HTMLElement): void {
    const form = document.createElement('form');
    form.className = 'form';

    this.fields.forEach(field => {
      const group = MonochromeComponents.createElement('div', {
        className: 'form-group'
      });

      const label = MonochromeComponents.createElement('label', {
        className: 'label',
        attributes: { for: field.name }
      }, [field.label]);

      group.appendChild(label);

      let input: HTMLElement;

      switch (field.type) {
        case 'textarea':
          input = MonochromeComponents.createElement('textarea', {
            className: 'textarea',
            id: field.name,
            attributes: { name: field.name }
          });
          break;
        case 'select':
          input = MonochromeComponents.createElement('select', {
            className: 'select',
            id: field.name,
            attributes: { name: field.name }
          });
          field.options?.forEach(option => {
            const optEl = MonochromeComponents.createElement('option', {
              attributes: { value: option.value }
            }, [option.label]);
            input.appendChild(optEl);
          });
          break;
        default:
          input = MonochromeComponents.input({
            type: field.type,
            id: field.name,
            onChange: (value) => this.setValue(field.name, value)
          });
      }

      group.appendChild(input);

      if (this.errors[field.name]) {
        const error = MonochromeComponents.createElement('span', {
          className: 'helper-text error'
        }, [this.errors[field.name]]);
        group.appendChild(error);
      }

      form.appendChild(group);
    });

    container.appendChild(form);
  }
}

// Usage

// monochrome-jquery.ts
/// <reference types="jquery" />

(function($: JQueryStatic) {
  // Main plugin
  $.fn.monochrome = function(this: JQuery, options: MonochromeOptions = {}) {
    const settings = $.extend({
      theme: 'warm',
      mode: 'light'
    }, options);

    return this.each(function() {
      const $element = $(this);

      // Apply theme
      $element.attr('data-theme', settings.mode);

      // Load theme CSS
      const themeLink = $('#theme-link');
      if (themeLink.length) {
        themeLink.attr('href', `ui/tokens/${settings.theme}-theme.css`);
      }
    });
  };

  // Button plugin
  $.fn.monochromeButton = function(this: JQuery, options: MonochromeButtonOptions = {}) {
    const settings = $.extend({
      variant: 'primary',
      size: 'medium',
      loading: false
    }, options);

    return this.each(function() {
      const $button = $(this);
      const sizeClass = settings.size === 'small' ? 'btn-small' :
                        settings.size === 'large' ? 'btn-large' : '';

      $button
        .addClass(`btn btn-${settings.variant} ${sizeClass}`)
        .toggleClass('loading', settings.loading || false);

      if (settings.loading) {
        $button.prop('disabled', true);
      }
    });
  };

  // Modal plugin
  $.fn.monochromeModal = function(this: JQuery, options: MonochromeModalOptions = {}) {
    const settings = $.extend({
      title: 'Modal Title',
      content: 'Modal content',
      size: 'medium',
      closeOnOverlay: true,
      onOpen: () => {},
      onClose: () => {}
    }, options);

    return this.each(function() {
      const $trigger = $(this);

      $trigger.on('click', function(e: JQuery.ClickEvent) {
        e.preventDefault();

        const modalHtml = `
          <div class="modal">
            <div class="modal-backdrop"></div>
            <div class="modal-content modal-${settings.size}">
              <div class="modal-header">
                <h3 class="modal-title">${settings.title}</h3>
                <button class="modal-close">√ó</button>
              </div>
              <div class="modal-body"></div>
              <div class="modal-footer">
                <button class="btn btn-ghost modal-cancel">Cancel</button>
                <button class="btn btn-primary modal-confirm">Confirm</button>
              </div>
            </div>
          </div>
        `;

        const $modal = $(modalHtml);

        // Add content
        if (typeof settings.content === 'string') {
          $modal.find('.modal-body').html(settings.content);
        } else {
          $modal.find('.modal-body').append(settings.content);
        }

        // Append to body
        $('body').append($modal);

        // Show modal
        setTimeout(() => {
          $modal.addClass('is-open');
          settings.onOpen?.();
        }, 10);

        // Close handlers
        const closeModal = () => {
          $modal.removeClass('is-open');
          setTimeout(() => {
            $modal.remove();
            settings.onClose?.();
          }, 300);
        };

        $modal.find('.modal-close, .modal-cancel').on('click', closeModal);

        if (settings.closeOnOverlay) {
          $modal.find('.modal-backdrop').on('click', closeModal);
        }
      });
    });
  };

  // Toast plugin
  $.fn.monochromeToast = function(message: string, type: 'success' | 'error' | 'info' = 'info') {
    const toastHtml = `
      <div class="toast toast-${type}">
        <div class="toast-content">
          <span class="toast-message">${message}</span>
          <button class="toast-close">√ó</button>
        </div>
      </div>
    `;

    const $toast = $(toastHtml);

    // Create container if not exists
    let $container = $('.toast-container');
    if (!$container.length) {
      $container = $('<div class="toast-container"></div>');
      $('body').append($container);
    }

    $container.append($toast);

    // Auto remove after 5 seconds
    setTimeout(() => {
      $toast.fadeOut(300, function() {
        $(this).remove();
      });
    }, 5000);

    // Manual close
    $toast.find('.toast-close').on('click', function() {
      $toast.fadeOut(300, function() {
        $(this).remove();
      });
    });

    return this;
  };
})(jQuery);

// components/BaseComponent.ts
export abstract class BaseComponent<T extends HTMLElement = HTMLElement> {
  protected element: T;
  protected options: Record<string, any>;

  constructor(element: T | string, options: Record<string, any> = {}) {
    this.element = typeof element === 'string'
      ? document.querySelector<T>(element)!
      : element;
    this.options = options;
    this.init();
  }

  protected abstract init(): void;

  protected addClass(...classes: string[]): void {
    this.element.classList.add(...classes.filter(Boolean));
  }

  protected removeClass(...classes: string[]): void {
    this.element.classList.remove(...classes);
  }

  protected toggleClass(className: string, force?: boolean): void {
    this.element.classList.toggle(className, force);
  }

  protected on<K extends keyof HTMLElementEventMap>(
    event: K,
    handler: (ev: HTMLElementEventMap[K]) => void
  ): void {
    this.element.addEventListener(event, handler);
  }

  protected off<K extends keyof HTMLElementEventMap>(
    event: K,
    handler: (ev: HTMLElementEventMap[K]) => void
  ): void {
    this.element.removeEventListener(event, handler);
  }

  protected emit(eventName: string, detail?: any): void {
    this.element.dispatchEvent(new CustomEvent(eventName, { detail }));
  }

  public destroy(): void {
    // Override in child classes for cleanup
  }
}

// components/Button/Button.ts
import { BaseComponent } from '../BaseComponent';
import { Size, Variant } from '../../tokens';

export interface ButtonOptions {
  variant?: Variant;
  size?: Size;
  loading?: boolean;
  disabled?: boolean;
  icon?: string;
}

export class Button extends BaseComponent<HTMLButtonElement> {
  private _loading = false;
  private _disabled = false;

  constructor(element: HTMLButtonElement | string, options: ButtonOptions = {}) {
    super(element, options);
  }

  protected init(): void {
    this.addClass('btn');
    this.setVariant(this.options.variant || 'primary');
    this.setSize(this.options.size || 'medium');

    if (this.options.loading) {
      this.setLoading(true);
    }

    if (this.options.disabled) {
      this.setDisabled(true);
    }

    this.on('click', this.handleClick.bind(this));
  }

  private handleClick(event: MouseEvent): void {
    if (this._loading || this._disabled) {
      event.preventDefault();
      event.stopPropagation();
      return;
    }
    this.emit('monochrome:click', { button: this });
  }

  public setVariant(variant: Variant): void {
    // Remove old variant classes
    ['primary', 'secondary', 'ghost', 'outline', 'danger'].forEach(v => {
      this.removeClass(`btn-${v}`);
    });
    this.addClass(`btn-${variant}`);
  }

  public setSize(size: Size): void {
    // Remove old size classes
    this.removeClass('btn-small', 'btn-large');
    if (size !== 'medium') {
      this.addClass(`btn-${size}`);
    }
  }

  public setLoading(loading: boolean): void {
    this._loading = loading;
    this.toggleClass('loading', loading);
    this.element.disabled = loading || this._disabled;

    if (loading) {
      this.element.setAttribute('aria-busy', 'true');
    } else {
      this.element.removeAttribute('aria-busy');
    }
  }

  public setDisabled(disabled: boolean): void {
    this._disabled = disabled;
    this.element.disabled = disabled || this._loading;

    if (disabled) {
      this.element.setAttribute('aria-disabled', 'true');
    } else {
      this.element.removeAttribute('aria-disabled');
    }
  }

  public get loading(): boolean {
    return this._loading;
  }

  public get disabled(): boolean {
    return this._disabled;
  }

  public static create(text: string, options: ButtonOptions = {}): Button {
    const element = document.createElement('button');
    element.textContent = text;
    return new Button(element, options);
  }
}

// Auto-initialize buttons with data attributes
document.addEventListener('DOMContentLoaded', () => {
  document.querySelectorAll<HTMLButtonElement>('[data-monochrome="button"]').forEach(el => {
    new Button(el, {
      variant: el.dataset.variant as Variant,
      size: el.dataset.size as Size,
      loading: el.dataset.loading === 'true',
      disabled: el.hasAttribute('disabled')
    });
  });

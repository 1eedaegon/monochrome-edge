Monochrome Edge UI Components

A modern, minimalist UI Components with two distinct themes. Warm theme for content management systems and publishing, Cold theme for SaaS applications and API businesses.

Warm Theme

Warm, content-focused design for CMS and publishing platforms. Features Blond Wood (#D3C8AB) as the primary accent.

Cold Theme

Cold, professional design for SaaS and API-driven businesses. Features Basalt (#8C9092) as the primary accent.

import React, { useState } from 'react';
import { FormGroup, Label, Input, Button } from '@monochrome-edge';  // TSX components

function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  return (
    <form onSubmit={(e) => e.preventDefault()}>
      <FormGroup>
        <Label htmlFor="email" required>Email</Label>
        <Input
          id="email"
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          placeholder="you@example.com"
        />
      </FormGroup>

      <FormGroup>
        <Label htmlFor="password" required>Password</Label>
        <Input
          id="password"
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
        />
      </FormGroup>

      <Button type="submit" variant="primary">
        Sign In
      </Button>
    </form>
  );
}

Integration Guide

How to integrate Monochrome Edge UI Components into your project.

Installation

Option 1: Yarn Package (Recommended)

# Install the core UI Components (includes React/TSX support)
yarn add @monochrome

# Install framework-specific packages
yarn add @monochrome-edge/vue       # For Vue
yarn add @monochrome-edge/vanilla   # For Vanilla JS/TS
yarn add @monochrome-edge/jquery    # For jQuery projects
yarn add @monochrome-edge/wc        # For Web Components

Option 2: CDN

<!-- CSS -->
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/@monochrome-edge/dist/monochrome-edge.css">
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/@monochrome-edge/dist/tokens/cold-theme.css">

<!-- JavaScript (optional, for components) -->
<script src="https://cdn.jsdelivr.net/npm/@monochrome-edge/vanilla/dist/index.js"></script>

Option 3: Local Development

# Clone the repository
git clone https://github.com/your-org/monochrome-edge.git

# Navigate to package directory
cd monochrome-edge/packages/[framework]

# Install dependencies
npm install

# Build the package
npm run build

# Link for local development
npm link

Step 3: Extend Monochrome components

// src/components/Button.ts
import '@monochrome-edge/dist/monochrome-edge.css';

export class EnhancedButton {
  private element: HTMLButtonElement;

  constructor(text: string, options?: any) {
    this.element = document.createElement('button');
    this.element.className = 'btn btn-primary';
    this.element.textContent = text;

    // Add custom functionality
    this.element.addEventListener('click', () => {
      this.element.classList.add('pulse-animation');
      setTimeout(() => {
        this.element.classList.remove('pulse-animation');
      }, 300);
    });
  }

  mount(container: HTMLElement) {
    container.appendChild(this.element);
  }
}

  animation: pulse 0.3s ease-in-out;

@keyframes pulse {
  0% { transform: scale(1); }
  50% { transform: scale(1.05); }
  100% { transform: scale(1); }
}

Step 5: Export your library

// src/index.ts
export { EnhancedButton } from './components/Button';
export { CustomCard } from './components/Card';

// Export a convenience initialization function
export function initializeCustomLibrary() {
  // Load Monochrome CSS if not already loaded
  if (!document.getElementById('monochrome-css')) {
    const link = document.createElement('link');
    link.id = 'monochrome-css';
    link.rel = 'stylesheet';
    link.href = 'node_modules/@monochrome-edge/dist/monochrome-edge.css';
    document.head.appendChild(link);
  }

  // Auto-enhance existing elements
  document.querySelectorAll('[data-custom]').forEach(el => {
    // Apply custom enhancements
  });
}

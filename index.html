Monochrome Edge UI Components

A modern, minimalist UI Components with two distinct themes. Warm theme for content management systems and publishing, Cold theme for SaaS applications and API businesses.

Warm Theme

Warm, content-focused design for CMS and publishing platforms. Features Blond Wood (#D3C8AB) as the primary accent.

Cold Theme

Cold, professional design for SaaS and API-driven businesses. Features Basalt (#8C9092) as the primary accent.

function toggleDropdown(button) {
    const dropdown = $(button).parent('.dropdown');
    dropdown.toggleClass('is-open');
    dropdown.find('.dropdown-panel').toggleClass('is-open');
}

$('.dropdown-option').on('click', function() {
    const text = $(this).text();
    $(this).closest('.dropdown').find('.dropdown-button span:first').text(text);
    $(this).siblings().removeClass('is-selected');
    $(this).addClass('is-selected');
    $(this).closest('.dropdown').removeClass('is-open');
    $(this).closest('.dropdown-panel').removeClass('is-open');
});

    this.preview = this.container.querySelector('.markdown-preview');
    this.editorContent = this.container.querySelector('.editor-content');
  }

    // Editor input
    this.editor.addEventListener('input', () => {
      this.content = this.editor.value;
      this.updatePreview();
    });

    // Toolbar buttons
    this.container.querySelectorAll('[data-format]').forEach(btn => {
      btn.addEventListener('click', (e) => {
        this.formatText(e.target.dataset.format);
      });
    });

    // View mode buttons
    this.container.querySelectorAll('[data-view]').forEach(btn => {
      btn.addEventListener('click', (e) => {
        this.setViewMode(e.target.dataset.view);
      });
    });
  }

  formatText(format) {
    const start = this.editor.selectionStart;
    const end = this.editor.selectionEnd;
    const selectedText = this.content.substring(start, end);

    let formatted;
    switch(format) {
      case 'bold':
        formatted = `**${selectedText}**`;
        break;
      case 'italic':
        formatted = `*${selectedText}*`;
        break;
      case 'heading':
        formatted = `# ${selectedText}`;
        break;
      default:
        formatted = selectedText;
    }

    this.content = this.content.substring(0, start) +
                  formatted +
                  this.content.substring(end);
    this.editor.value = this.content;
    this.updatePreview();

    // Restore cursor position
    const newPos = start + formatted.length;
    this.editor.setSelectionRange(newPos, newPos);
    this.editor.focus();
  }

  setViewMode(mode) {
    this.viewMode = mode;
    this.editorContent.className = `editor-content ${mode}-mode`;

    // Update button states
    this.container.querySelectorAll('[data-view]').forEach(btn => {
      btn.className = btn.dataset.view === mode ? 'btn-primary' : 'btn-ghost';
    });
  }

  updatePreview() {
    this.preview.innerHTML = marked(this.content);
  }

  getContent() {
    return this.content;
  }

  setContent(content) {
    this.content = content;
    this.editor.value = content;
    this.updatePreview();
  }
}

            break;
          case 'italic':

            break;

            formatted = `# ${selectedText}`;
            break;
          default:
            formatted = selectedText;

        $(textarea).val(content);
        updatePreview();

        // Restore cursor position
        const newPos = start + formatted.length;
        textarea.setSelectionRange(newPos, newPos);
        textarea.focus();
      }

          .removeClass('editor-mode preview-mode split-mode')
          .addClass(mode + '-mode');

          $(this).removeClass('btn-primary btn-ghost')
            .addClass($(this).data('view') === mode ? 'btn-primary' : 'btn-ghost');
        });
      }

      render();
      updatePreview();

      // Event listeners

        content = $(this).val();
        updatePreview();
      });

      $container.on('click', '[data-format]', function() {
        formatText($(this).data('format'));
      });

      $container.on('click', '[data-view]', function() {
        setViewMode($(this).data('view'));
      });

        getContent: function() { return content; },
        setContent: function(newContent) {
          content = newContent;

          updatePreview();
        },
